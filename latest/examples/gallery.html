    Examples converted from the PGFPlots manual gallery. This is a work in progress. All the examples are run with the following code added to them.

    import PGFPlotsX
const pgf = PGFPlotsX; # hide
using LaTeXStrings
    pgf.@pgf pgf.Axis(
    {
        xlabel = "Cost",
        ylabel = "Error",
    },
    pgf.Plot(
        {
            color = "red",
            mark  = "x"
        },
        pgf.Coordinates(
            [
                (2, -2.8559703),
                (3, -3.5301677),
                (4, -4.3050655),
                (5, -5.1413136),
                (6, -6.0322865),
                (7, -6.9675052),
                (8, -7.9377747),
            ]
        ),
    ),
)
    pgf.@pgf pgf.Axis(
    {
        xlabel = L"x",
        ylabel = L"f(x) = x^2 - x + 4"
    },
    pgf.Plot(
        pgf.Expression("x^2 - x + 4")
    )
)
